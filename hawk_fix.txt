# Commands to run on HAWK server to fix SSH banner issue:

# 1. Check current sshd_config for Banner directive
Get-Content C:\ProgramData\ssh\sshd_config | Select-String -Pattern 'Banner'

# 2. Remove Banner directive if it exists
$config = Get-Content C:\ProgramData\ssh\sshd_config
$config = $config | Where-Object { $_ -notmatch '^Banner' }
$config | Set-Content C:\ProgramData\ssh\sshd_config -Encoding ASCII

# 3. Restart SSH service
Restart-Service sshd

# 4. Test locally
ssh alleycat@localhost 'echo test'
